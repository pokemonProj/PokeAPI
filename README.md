```json

          "name": "limber",

       },
       "is_hidden": false,
       "slot": 1
    },
    {
       "ability": 
       {
          "name": "imposter",
          "url": "https://pokeapi.co/api/v2/ability/150/"

       "slot": 3
    }
] 
```
